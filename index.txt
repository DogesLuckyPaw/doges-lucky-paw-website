        // Function to copy contract address to clipboard
        function copyContractAddress() {
            const contractAddress = document.getElementById('contract-address').innerText;
            const tempInput = document.createElement('textarea');
            tempInput.value = contractAddress;
            document.body.appendChild(tempInput);
            tempInput.select();
            document.execCommand('copy');

        // Function to format numbers with commas for thousands
        function formatNumberWithCommas(value) {
            if (value === null || isNaN(value)) {
                return 'N/A';
            }
            return new Intl.NumberFormat('en-US', {
                minimumFractionDigits: 0,
                maximumFractionDigits: 0
            }).format(value);
        }

        // Function to format numbers as currency
        function formatCurrency(value) {
            if (value === null || isNaN(value)) {
                return 'N/A';
            }
            return new Intl.NumberFormat('en-US', {
                style: 'currency',
                currency: 'USD',
                minimumFractionDigits: 2,
                maximumFractionDigits: 2
            }).format(value);
        }

        // Function to fetch and update market data from your Cloudflare Worker
        async function updateMarketData() {
            const bondingCurveProgressElement = document.getElementById('bonding-curve-progress-value');
            const liveMarketCapElement = document.getElementById('live-market-cap-value');
            const liquidityElement = document.getElementById('liquidity-value');
            const circulatingSupplyElement = document.getElementById('circulating-supply-value');
            const lastUpdatedElement = document.getElementById('last-updated');
            
            // =====================================================================
            // IMPORTANT: This URL now points to your deployed Cloudflare Worker.
            // =====================================================================
            const WORKER_URL = 'https://doges-lucky-paw-backend.pfeifferjohnathon.workers.dev/api/market-data';
            const localStorageKey = 'dogePawMarketData';

            // Function to display data on the page
            const displayData = (data) => {
                if (!data) return;

                const {bondingCurveProgress, liveMarketCap, liquidity, circulatingSupply, lastUpdated} = data;
                
                bondingCurveProgressElement.textContent = !isNaN(bondingCurveProgress) ? `${bondingCurveProgress.toFixed(2)}%` : 'N/A';
                liveMarketCapElement.textContent = !isNaN(liveMarketCap) ? formatCurrency(liveMarketCap) : 'N/A';
                liquidityElement.textContent = !isNaN(liquidity) ? formatCurrency(liquidity) : 'N/A';
                circulatingSupplyElement.textContent = !isNaN(circulatingSupply) ? `${formatNumberWithCommas(circulatingSupply)} $PAW` : 'N/A';
                lastUpdatedElement.textContent = lastUpdated ? `Last updated: ${new Date(lastUpdated).toLocaleString()}` : '';
            };

            // First, try to load data from local storage
            const cachedData = localStorage.getItem(localStorageKey);
            if (cachedData) {
                displayData(JSON.parse(cachedData));
            } else {
                // If no cached data, show initial loading state
                bondingCurveProgressElement.textContent = 'Loading...';
                liveMarketCapElement.textContent = 'Loading...';
                liquidityElement.textContent = 'Loading...';
                circulatingSupplyElement.textContent = 'Loading...';
            }

            let retries = 0;
            const maxRetries = 5;
            let delay = 1000;

            while (retries < maxRetries) {
                try {
                    const response = await fetch(WORKER_URL); // Fetch from your Worker
                    
                    if (!response.ok) {
                        const errorDetails = await response.json();
                        throw new Error(`Worker error! Status: ${response.status}, Message: ${errorDetails.error || 'Unknown error'}`);
                    }

                    const data = await response.json();
                    displayData(data);
                    localStorage.setItem(localStorageKey, JSON.stringify(data));
                    return; // Success, exit loop

                } catch (error) {
                    console.error("Error fetching market data from Worker:", error);
                    if (retries === maxRetries - 1) {
                        console.error("Failed to fetch market data after multiple retries.");
                        // Optionally display a user-friendly error message on the page
                    }
                    retries++;
                    await new Promise(resolve => setTimeout(resolve, delay));
                    delay *= 2;
                }
            }
        }

        // Call the function when the page loads
        window.onload = updateMarketData;
    </script>
